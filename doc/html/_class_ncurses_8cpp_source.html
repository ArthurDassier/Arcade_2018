121 void ClassNcurses::displayGame()
122 {

124  std::string wall;

131  size_t posi = 1;
132 
133  for (auto it = parsingResult.begin(); it != parsingResult.end(); ++it) {
134  switch (i) {
135  case 1:
136  wall = it->caractere;
137  break;
138  case 2:
139  miam = it->caractere;
140  break;
141  case 3:
142  perso = it->caractere;
143  break;
144  case 4:
145  enemy = it->caractere;
146  break;
147  case 5:
148  super_miam = it->caractere;
149  break;
150  default:
151  break;
152  }
153  i++;
154  }
155 
156  if (wall.compare("sfml") == false) {
157  displayMenu();
158  displayLetters();
159  return;
160  }
161 
162  init_pair(1, COLOR_BLUE, COLOR_BLACK);
163  init_pair(2, COLOR_YELLOW, COLOR_BLACK);
164  init_pair(3, COLOR_BLUE, COLOR_WHITE);
165  move((LINES/2) - (_map->size()/2), (COLS/2) - (_map->begin()->size()/2)); //dimensions de la map
166  for (auto it = _map->begin(); it != _map->end(); ++it) {
167  for (auto i = it->begin(); i != it->end(); ++i) {
168  switch (*i) {
169  case NOTHING: printw(" ");
170  break;
171  case WALL: attron(COLOR_PAIR(1));
172  if (it == _map->begin() || ((it + 1) == _map->end()))
173  printw("-");
174  else if (i == it->begin() || ((i + 1) == it->end()))
175  printw("|");
176  else if (lock_wall != 0) {
177  printw("|");
178  lock_wall = 0;
179  } else
180  printw("%s", wall.c_str());
181  attroff(COLOR_PAIR(1));
182  break;
183  case POINT: printw("%s", miam.c_str());
184  break;
185  case PLAYER: attron(COLOR_PAIR(2));
186  printw("%s", perso.c_str());
187  attroff(COLOR_PAIR(2));
188  break;
189  case GHOST: printw("%s", enemy.c_str());
190  break;
191  case BONUS: printw("%s", super_miam.c_str());
192  break;
193  default: printw(" ");
194  if (i == it->begin())
195  lock_wall++;
196  break;
197  }
198  }
199  printw("\n");
200  move((LINES/2) - (_map->size()/2) + posi, (COLS/2) - (_map->begin()->size()/2));
201  posi++;
202  }
203 }
