156 void ClassNcurses::displayGame()
157 {
158  std::vector<DataParsingConfig> parsingResult = _parsing.getResult();
159  std::string wall;
160  std::string miam;
161  std::string perso;
162  std::string enemy;
163  std::string super_miam;
164  int i = 0;
165  size_t lock_wall = 0;
166  size_t posi = 1;
167 
168  for (auto it = parsingResult.begin(); it != parsingResult.end(); ++it) {
169  switch (i) {
170  case 1:
171  wall = it->caractere;
172  break;
173  case 2:
174  miam = it->caractere;
175  break;
176  case 3:
177  perso = it->caractere;
178  break;
179  case 4:
180  enemy = it->caractere;
181  break;
182  case 5:
183  super_miam = it->caractere;
184  break;
185  default:
186  break;
187  }
188  i++;
189  }
190 
191  if (wall.compare("sfml") == false) {
192  displayMenu();
193  displayLetters();
194  return;
195  }
196 
197  init_pair(1, COLOR_BLUE, COLOR_BLACK);
198  init_pair(2, COLOR_YELLOW, COLOR_BLACK);
199  init_pair(3, COLOR_BLUE, COLOR_WHITE);
200  move((LINES/2) - (_map->size()/2), (COLS/2) - (_map->begin()->size()/2)); //dimensions de la map
201  for (auto it = _map->begin(); it != _map->end(); ++it) {
202  for (auto i = it->begin(); i != it->end(); ++i) {
203  switch (*i) {
204  case NOTHING: printw(" ");
205  break;
206  case WALL: attron(COLOR_PAIR(1));
207  if (it == _map->begin() || ((it + 1) == _map->end()))
208  printw("-");
209  else if (i == it->begin() || ((i + 1) == it->end()))
210  printw("|");
211  else if (lock_wall != 0) {
212  printw("|");
213  lock_wall = 0;
214  } else
215  printw("%s", wall.c_str());
216  attroff(COLOR_PAIR(1));
217  break;
218  case POINT: printw("%s", miam.c_str());
219  break;
220  case PLAYER: attron(COLOR_PAIR(2));
221  printw("%s", perso.c_str());
222  attroff(COLOR_PAIR(2));
223  break;
224  case GHOST: printw("%s", enemy.c_str());
225  break;
226  case BONUS: printw("%s", super_miam.c_str());
227  break;
228  default: printw(" ");
229  if (i == it->begin())
230  lock_wall++;
231  break;
232  }
233  }
234  printw("\n");
235  move((LINES/2) - (_map->size()/2) + posi, (COLS/2) - (_map->begin()->size()/2));
236  posi++;
237  }
238 }
