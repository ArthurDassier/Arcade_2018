102 void AnimatedSprite::setFrame(std::size_t newFrame, bool resetTime)
103 {

107 
108  m_vertices[0].position = sf::Vector2f(0.f, 0.f);

110  m_vertices[2].position = sf::Vector2f(static_cast<float>(rect.width), static_cast<float>(rect.height));

112 
113  float left = static_cast<float>(rect.left) + 0.0001f;
114  float right = left + static_cast<float>(rect.width);
115  float top = static_cast<float>(rect.top);

120  m_vertices[2].texCoords = sf::Vector2f(right, bottom);

127 void AnimatedSprite::update(sf::Time deltaTime)
128 {
129  // if not paused and we have a valid animation
130  if (!m_isPaused && m_animation) {
131  // add delta time
132  m_currentTime += deltaTime;
133  // if current time is bigger then the frame time advance one frame
134  if (m_currentTime >= m_frameTime) {
135  // reset time, but keep the remainder
136  m_currentTime = sf::microseconds(m_currentTime.asMicroseconds() % m_frameTime.asMicroseconds());
137 
138  // get next Frame index
139  if (m_currentFrame + 1 < m_animation->getSize())
140  m_currentFrame++;
141  else {
142  // animation has ended
143  m_currentFrame = 0; // reset to start
144  if (!m_isLooped)
145  m_isPaused = true;
146  }
147  // set the current frame, not reseting the time
148  setFrame(m_currentFrame, false);
149  }
150  }
151 }
